CREATE TABLE businesses (
    businesses_id SERIAL PRIMARY KEY,
    businesses_unique_id VARCHAR(255) DEFAULT NULL UNIQUE,
    businesses_user_name VARCHAR(255) DEFAULT NULL UNIQUE,
    businesses_legal_name VARCHAR(255) DEFAULT NULL UNIQUE,
    businesses_organization_name VARCHAR(255) DEFAULT NULL UNIQUE,
    businesses_gst_number VARCHAR(255) DEFAULT NULL UNIQUE,
    businesses_website VARCHAR(255) DEFAULT NULL,
    businesses_documents TEXT DEFAULT NULL,
    businesses_logo TEXT DEFAULT NULL,
    businesses_is_active BOOLEAN DEFAULT TRUE,
    businesses_is_deleted BOOLEAN DEFAULT FALSE,
    businesses_created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    businesses_updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
